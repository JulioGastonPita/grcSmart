> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Construir la API REST en Go con Gin que gestiona documentos PDF, embeddings en ChromaDB y consultas RAG vía Gemini.

**Architecture:** Arquitectura hexagonal ligera: `cmd/server` (entrypoint) → `internal/api` (handlers + router) → `internal/domain` (modelos) → `internal/infrastructure` (postgres, chromadb, gemini, storage). Los handlers no contienen lógica de negocio; esta vive en servicios en `internal/service`.

**Tech Stack:** Go 1.22, Gin v1.10, GORM + driver postgres, amikos-tech/chroma-go v0.4, google/generative-ai-go v0.13, ledongthuc/pdf v0.0.0-20210923183342, golang.org/x/text.

```go
package storage

import (
	"fmt"
	"os"
	"path/filepath"
	"time"
)

// Save guarda content en uploadsDir con un nombre único basado en timestamp + nombre original.
// Retorna la ruta absoluta del archivo guardado.
func Save(uploadsDir, originalName string, content []byte) (string, error) {
	if err := os.MkdirAll(uploadsDir, 0755); err != nil {
		return "", fmt.Errorf("creating uploads dir: %w", err)
	}

	filename := fmt.Sprintf("%d_%s", time.Now().UnixNano(), originalName)
	path := filepath.Join(uploadsDir, filename)

	if err := os.WriteFile(path, content, 0644); err != nil {
		return "", fmt.Errorf("writing file: %w", err)
	}

	return path, nil
}

// Delete elimina el archivo en la ruta dada.
func Delete(path string) error {
	if err := os.Remove(path); err != nil && !os.IsNotExist(err) {
		return fmt.Errorf("deleting file %s: %w", path, err)
	}
	return nil
}
```

```go
package service

import (
	"bytes"
	"fmt"

	"github.com/ledongthuc/pdf"
)

// ExtractTextFromPDF extrae todo el texto de un PDF dado como bytes.
func ExtractTextFromPDF(data []byte) (string, error) {
	r, err := pdf.NewReader(bytes.NewReader(data), int64(len(data)))
	if err != nil {
		return "", fmt.Errorf("opening pdf: %w", err)
	}

	var buf bytes.Buffer
	for i := 1; i <= r.NumPage(); i++ {
		p := r.Page(i)
		if p.V.IsNull() {
			continue
		}
		text, err := p.GetPlainText(nil)
		if err != nil {
			continue // página sin texto legible
		}
		buf.WriteString(text)
		buf.WriteString(" ")
	}

	return buf.String(), nil
}

// ChunkText divide text en chunks de tamaño chunkSize con overlap entre ellos.
// Algoritmo Recursive Character Text Splitter simplificado.
func ChunkText(text string, chunkSize, overlap int) []string {
	if len(text) <= chunkSize {
		return []string{text}
	}

	var chunks []string
	start := 0

	for start < len(text) {
		end := start + chunkSize
		if end > len(text) {
			end = len(text)
		}
		chunks = append(chunks, text[start:end])
		next := start + chunkSize - overlap
		if next <= start {
			break
		}
		start = next
	}

	return chunks
}
```

```go
package llm

import (
	"context"
	"fmt"

	"google.golang.org/genai"
)

// GeminiClient implementa Embedder y LLMClient usando la API de Gemini.
type GeminiClient struct {
	client    *genai.Client
	llmModel  string
	embedModel string
}

// NewGeminiClient crea un cliente Gemini con la API key dada.
func NewGeminiClient(ctx context.Context, apiKey, llmModel string) (*GeminiClient, error) {
	client, err := genai.NewClient(ctx, &genai.ClientConfig{
		APIKey:  apiKey,
		Backend: genai.BackendGeminiAPI,
	})
	if err != nil {
		return nil, fmt.Errorf("creating gemini client: %w", err)
	}

	return &GeminiClient{
		client:     client,
		llmModel:   llmModel,
		embedModel: "text-embedding-004",
	}, nil
}

// Embed genera un embedding para el texto dado.
func (g *GeminiClient) Embed(ctx context.Context, text string) ([]float32, error) {
	result, err := g.client.Models.EmbedContent(ctx, g.embedModel,
		genai.NewContentFromText(text, genai.RoleUser), nil)
	if err != nil {
		return nil, fmt.Errorf("embedding text: %w", err)
	}
	return result.Embeddings[0].Values, nil
}

// Complete envía un prompt al LLM y retorna la respuesta de texto.
func (g *GeminiClient) Complete(ctx context.Context, systemPrompt, userMessage string) (string, error) {
	model := g.client.Models
	config := &genai.GenerateContentConfig{
		SystemInstruction: genai.NewContentFromText(systemPrompt, genai.RoleUser),
	}

	resp, err := model.GenerateContent(ctx, g.llmModel,
		genai.NewContentFromText(userMessage, genai.RoleUser), config)
	if err != nil {
		return "", fmt.Errorf("generating content: %w", err)
	}

	if len(resp.Candidates) == 0 || len(resp.Candidates[0].Content.Parts) == 0 {
		return "", fmt.Errorf("empty response from gemini")
	}

	part := resp.Candidates[0].Content.Parts[0]
	if text, ok := part.(genai.Text); ok {
		return string(text), nil
	}
	return "", fmt.Errorf("unexpected response part type")
}
```

```go
package vectordb

import (
	"context"
	"fmt"

	chroma "github.com/amikos-tech/chroma-go"
	"github.com/amikos-tech/chroma-go/collection"
	chromatypes "github.com/amikos-tech/chroma-go/types"
)

const collectionName = "grcsmart_documents"

// ChromaStore gestiona el almacenamiento y búsqueda de vectores en ChromaDB.
type ChromaStore struct {
	client     *chroma.Client
	collection *collection.Collection
}

// NewChromaStore crea o reutiliza la colección en ChromaDB.
func NewChromaStore(ctx context.Context, chromaURL string) (*ChromaStore, error) {
	client, err := chroma.NewClient(chroma.WithBasePath(chromaURL))
	if err != nil {
		return nil, fmt.Errorf("creating chroma client: %w", err)
	}

	col, err := client.NewCollection(
		ctx,
		collectionName,
		map[string]any{},
		chromatypes.L2,
	)
	if err != nil {
		// Si ya existe, obtenemos la colección existente
		col, err = client.GetCollection(ctx, collectionName, nil)
		if err != nil {
			return nil, fmt.Errorf("getting chroma collection: %w", err)
		}
	}

	return &ChromaStore{client: client, collection: col}, nil
}

// AddDocumentChunks inserta chunks de texto con sus embeddings y metadatos.
func (s *ChromaStore) AddDocumentChunks(ctx context.Context, documentID uint, originalName string, chunks []string, embeddings [][]float32) error {
	ids := make([]string, len(chunks))
	metas := make([]map[string]any, len(chunks))

	for i, chunk := range chunks {
		ids[i] = fmt.Sprintf("doc_%d_chunk_%d", documentID, i)
		metas[i] = map[string]any{
			"document_id":   documentID,
			"original_name": originalName,
			"chunk_index":   i,
			"text":          chunk,
		}
	}

	_, err := s.collection.Add(ctx, embeddings, metas, nil, ids)
	if err != nil {
		return fmt.Errorf("adding chunks to chroma: %w", err)
	}

	return nil
}

// SearchSimilar busca los k chunks más similares al embedding dado.
// Retorna los textos de los chunks encontrados.
func (s *ChromaStore) SearchSimilar(ctx context.Context, embedding []float32, k int) ([]string, error) {
	results, err := s.collection.Query(ctx, [][]float32{embedding}, int32(k), nil, nil, nil)
	if err != nil {
		return nil, fmt.Errorf("querying chroma: %w", err)
	}

	var texts []string
	if len(results.Metadatas) > 0 {
		for _, meta := range results.Metadatas[0] {
			if text, ok := meta["text"].(string); ok {
				texts = append(texts, text)
			}
		}
	}

	return texts, nil
}

// DeleteDocumentChunks elimina todos los chunks asociados a un documentID.
func (s *ChromaStore) DeleteDocumentChunks(ctx context.Context, documentID uint) error {
	// Primero buscamos los IDs de los chunks de este documento
	where := map[string]any{"document_id": documentID}
	results, err := s.collection.Get(ctx, nil, where, nil, nil)
	if err != nil {
		return fmt.Errorf("getting chunks for document %d: %w", documentID, err)
	}

	if len(results.Ids) == 0 {
		return nil // No hay nada que borrar
	}

	_, err = s.collection.Delete(ctx, results.Ids, nil, nil)
	if err != nil {
		return fmt.Errorf("deleting chunks from chroma: %w", err)
	}

	return nil
}
```

```go
package service

import (
	"context"
	"fmt"

	"github.com/JulioGastonPita/grcSmart/backend/internal/domain"
	"github.com/JulioGastonPita/grcSmart/backend/internal/infrastructure/llm"
	"github.com/JulioGastonPita/grcSmart/backend/internal/infrastructure/storage"
	"github.com/JulioGastonPita/grcSmart/backend/internal/infrastructure/vectordb"
	"gorm.io/gorm"
)

const (
	chunkSize    = 1000
	chunkOverlap = 200
	topK         = 5
)

// DocumentService gestiona el ciclo de vida de documentos PDF en el sistema.
type DocumentService struct {
	db          *gorm.DB
	vectorStore *vectordb.ChromaStore
	embedder    llm.Embedder
	uploadsDir  string
}

func NewDocumentService(db *gorm.DB, vs *vectordb.ChromaStore, embedder llm.Embedder, uploadsDir string) *DocumentService {
	return &DocumentService{db: db, vectorStore: vs, embedder: embedder, uploadsDir: uploadsDir}
}

// Upload guarda el PDF en disco, registra en Postgres, extrae texto, chunkea y almacena embeddings.
func (s *DocumentService) Upload(ctx context.Context, originalName string, fileContent []byte) (*domain.Document, error) {
	// 1. Guardar en disco
	path, err := storage.Save(s.uploadsDir, originalName, fileContent)
	if err != nil {
		return nil, fmt.Errorf("saving file: %w", err)
	}

	// 2. Registrar en PostgreSQL
	doc := &domain.Document{OriginalName: originalName, StoragePath: path}
	if err := s.db.Create(doc).Error; err != nil {
		storage.Delete(path) // rollback archivo
		return nil, fmt.Errorf("saving document to db: %w", err)
	}

	// 3. Extraer texto del PDF
	text, err := ExtractTextFromPDF(fileContent)
	if err != nil {
		return doc, fmt.Errorf("extracting text (document saved but not indexed): %w", err)
	}

	// 4. Chunking
	chunks := ChunkText(text, chunkSize, chunkOverlap)

	// 5. Generar embeddings para cada chunk
	embeddings := make([][]float32, 0, len(chunks))
	for _, chunk := range chunks {
		emb, err := s.embedder.Embed(ctx, chunk)
		if err != nil {
			return doc, fmt.Errorf("embedding chunk: %w", err)
		}
		embeddings = append(embeddings, emb)
	}

	// 6. Almacenar en ChromaDB
	if err := s.vectorStore.AddDocumentChunks(ctx, doc.ID, originalName, chunks, embeddings); err != nil {
		return doc, fmt.Errorf("storing in vector db: %w", err)
	}

	return doc, nil
}

// Delete elimina el documento de ChromaDB, PostgreSQL y disco.
func (s *DocumentService) Delete(ctx context.Context, id uint) error {
	var doc domain.Document
	if err := s.db.First(&doc, id).Error; err != nil {
		return fmt.Errorf("document %d not found: %w", id, err)
	}

	// 1. Eliminar chunks de ChromaDB
	if err := s.vectorStore.DeleteDocumentChunks(ctx, id); err != nil {
		return fmt.Errorf("deleting from vector db: %w", err)
	}

	// 2. Eliminar archivo físico
	if err := storage.Delete(doc.StoragePath); err != nil {
		return fmt.Errorf("deleting file: %w", err)
	}

	// 3. Eliminar de PostgreSQL (soft delete via GORM)
	if err := s.db.Delete(&doc).Error; err != nil {
		return fmt.Errorf("deleting from db: %w", err)
	}

	return nil
}

// ListAll retorna todos los documentos no eliminados.
func (s *DocumentService) ListAll() ([]domain.Document, error) {
	var docs []domain.Document
	if err := s.db.Find(&docs).Error; err != nil {
		return nil, fmt.Errorf("listing documents: %w", err)
	}
	return docs, nil
}
```

```go
package service

import (
	"context"
	"fmt"
	"strings"

	"github.com/JulioGastonPita/grcSmart/backend/internal/infrastructure/llm"
	"github.com/JulioGastonPita/grcSmart/backend/internal/infrastructure/vectordb"
)

const systemPromptTemplate = `Eres un asistente experto en políticas y procedimientos corporativos.
Responde SOLO basándote en el siguiente contexto extraído de los documentos oficiales.
Si la información no está en el contexto, responde que no tienes información suficiente.

CONTEXTO:
%s`

// ChatService gestiona las consultas RAG.
type ChatService struct {
	vectorStore *vectordb.ChromaStore
	embedder    llm.Embedder
	llmClient   llm.LLMClient
}

func NewChatService(vs *vectordb.ChromaStore, embedder llm.Embedder, llmClient llm.LLMClient) *ChatService {
	return &ChatService{vectorStore: vs, embedder: embedder, llmClient: llmClient}
}

// Query realiza una búsqueda RAG: embedea la pregunta, recupera contexto y genera respuesta.
func (s *ChatService) Query(ctx context.Context, question string) (string, error) {
	// 1. Embedear la pregunta
	queryEmbedding, err := s.embedder.Embed(ctx, question)
	if err != nil {
		return "", fmt.Errorf("embedding question: %w", err)
	}

	// 2. Recuperar chunks relevantes
	contextChunks, err := s.vectorStore.SearchSimilar(ctx, queryEmbedding, topK)
	if err != nil {
		return "", fmt.Errorf("searching similar chunks: %w", err)
	}

	if len(contextChunks) == 0 {
		return "No encontré información relevante en los documentos cargados.", nil
	}

	// 3. Construir prompt con contexto
	context := strings.Join(contextChunks, "\n\n---\n\n")
	systemPrompt := fmt.Sprintf(systemPromptTemplate, context)

	// 4. Generar respuesta
	answer, err := s.llmClient.Complete(ctx, systemPrompt, question)
	if err != nil {
		return "", fmt.Errorf("generating answer: %w", err)
	}

	return answer, nil
}
```

```go
func CORS() gin.HandlerFunc {
	return func(c *gin.Context) {
		c.Header("Access-Control-Allow-Origin", "*")
		c.Header("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS")
		c.Header("Access-Control-Allow-Headers", "Content-Type, Authorization")

		if c.Request.Method == "OPTIONS" {
			c.AbortWithStatus(204)
			return
		}

		c.Next()
	}
}
```

```go
	}
```

```go
package api

import (
	"github.com/JulioGastonPita/grcSmart/backend/internal/api/handlers"

func NewRouter(db *gorm.DB, docSvc *service.DocumentService, chatSvc *service.ChatService) *gin.Engine {

	settingsH := handlers.NewSettingsHandler(db)
	documentsH := handlers.NewDocumentsHandler(docSvc)
	chatH := handlers.NewChatHandler(chatSvc)

	api := r.Group("/api")
	{
		api.GET("/health", func(c *gin.Context) { c.JSON(200, gin.H{"status": "ok"}) })

		api.GET("/settings", settingsH.Get)
		api.PUT("/settings", settingsH.Update)

		api.GET("/documents", documentsH.List)
		api.POST("/documents", documentsH.Upload)
		api.DELETE("/documents/:id", documentsH.Delete)

		api.POST("/chat", chatH.Query)
	}

	return r
}
```
